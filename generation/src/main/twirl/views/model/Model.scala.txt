@(m: codegen.model.Model)

@import codegen.model
@import views.txt.util
@import views.txt.model.identifiable

@hmm(x: model.Identifiable) = {@x match {
  case x: model.Thing => { @identifiable.Thing(x.asInstanceOf[model.Thing]) }
  case x: model.Operation => { @identifiable.Operation(x.asInstanceOf[model.Operation]) }
  case _ => { "None" }
}
}

case class @m.className (
  items: List[Identifiable] =
    @util.List(m.items.map(x => hmm(x)))
  )
